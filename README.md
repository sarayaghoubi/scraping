# scraping
sample web scraping tool for https://www.tgju.org/ website to fetch historical financial data

# acceptbale symbols:
  -	Dollar: 'price_dollar_rl', 
  -	Gold : 'mesghal','geram18','
  -	Euro : 'price_eur',
  -	GBP : 'price_gbp'
  -	AED : 'price_aed'
  - Bahar Coin : 'sekeb'

# libraries have been used:
  - beautifulsoup
  - selenium
  - request
  - psycopg2
  - pandas
  - mplfinance
